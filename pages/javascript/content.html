Extract

Strings

slice(start, end)

slice(start)

  • end not included in the extracted string.
  • If a parameter is negative, the position is counted from the end of the string.
  • If end is omitted returns the rest of the string.

let x2 = str.slice(0, -6); // 012
let x3 = str.slice(7); // 789

Replace

.replace("old", "new")

  • Case sensitive.
  • Replaces only the first match.
  • Returns a new string.
  • Regex are written without quotes.

let str1 = "Apple, Banana, Kiwi";
let str2 = str.replace("Banana", "Melon");
// "Apple, Melon, Kiwi"

Case insensitive regex:


let str3 = str.replace(/BANANA/i, "Melon");

Replace all regex:


let str4 = str.replace(/Banana/g, "Melon");

Functions: arguments

Contains all the arguments passed to the function

  • ES5
  • Array-like object accessible inside functions
  • For non-arrow functions
  • It's not an array
  • Can use arguments.length

function myConcat(separator) {
    let args = Array.prototype.slice.call(arguments, 1);
    return args.join(separator);
}

myConcat(', ', 'red', 'orange', 'blue');
// "red, orange, blue"

Functions: ...restParam

Contans the remaining arguments passed to the function

  • ES6
  • Must be prefixed by ...
  • Must be the last parameter
  • Must be declared
  • It's a standard array

function myFun(a,  b, ...manyMoreArgs) {
    console.log("a", a)
    console.log("b", b)
    console.log("manyMoreArgs", manyMoreArgs)
}
              
myFun("one", "two", "three", "four", "five", "six")
// a, one
// b, two
// manyMoreArgs, ["three", "four", "five", "six"]

class Waifu extends Summoner {
    constructor(name, age, aeons) {
        super(aeons);
        this.name = name;
        this.age = age;
    }
       
    bio() {	
        alert('Name: ' + this.name 
            + '\nAge: ' + this.age 
            + '\nAeons: ' + this.aeons);
    }
      
    get aeons() { 
        return this._aeons;
    }
      
    set aeons(_aeons) {	
        this._aeons = _aeons;
    }      			
}
